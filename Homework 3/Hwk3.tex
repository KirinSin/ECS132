			  &= E(X^3 - p^3 + 3(p^2) \cdot x - 3p \cdot (x^2))\\
			  &= E(X^3) - E(p^3) + E(3\cdot(p^2)\cdot X) - E(3p\cdot X^2)\\
			  &= E(X) - p^3 + 3(p^2)E(X) - 3p\cdot E(X^2)\\
			  &= p - p^3 + 3p^3 - 3p^2\\

\end{align*}
